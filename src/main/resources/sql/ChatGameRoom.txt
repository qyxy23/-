CREATE TABLE `chat_games` (
  `room_id` VARCHAR(36) NOT NULL COMMENT '房间唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤唯一ID（UUID）',
  `room_name` VARCHAR(255) NOT NULL COMMENT '房间名称',
  `creator_id` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID（关联sys_user）',
  `required_members` INT UNSIGNED NOT NULL DEFAULT 2 COMMENT '所需人数（至少2人）',
  `current_members` INT UNSIGNED NOT NULL DEFAULT 1 COMMENT '当前人数',
  `status` ENUM('WAITING', 'ACTIVE', 'FINISHED','CANCELLED') NOT NULL DEFAULT 'WAITING' COMMENT '状态：等待集齐/进行中/已结束/已取消',
  `create_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `update_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',
  `start_time` DATETIME(6) NULL COMMENT '游戏开始时间（游戏正式启动时填充）',
  `end_time` DATETIME(6) NULL COMMENT '游戏结束时间（游戏正常结束或取消时填充）',
  `group_avatar` VARCHAR(255) DEFAULT '' COMMENT '海龟汤图像URL',
  `privacy_type` ENUM('PUBLIC', 'PRIVATE') NOT NULL DEFAULT 'PUBLIC' COMMENT '房间隐私类型：PUBLIC-公开/PRIVATE-私密',
  `need_invite` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否需要邀请加入（仅私密房间有效）',
  PRIMARY KEY (`room_id`),
  INDEX `idx_creator_id` (`creator_id`),
  CONSTRAINT `fk_chat_games_creator`
    FOREIGN KEY (`creator_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='群聊房间表';


CREATE TABLE IF NOT EXISTS `chat_game_invitations` (
  `invitation_id` VARCHAR(36) NOT NULL COMMENT '邀请唯一ID（UUID）',
  `room_id` VARCHAR(36) NOT NULL COMMENT '目标房间ID（关联chat_games.room_id）',
  `inviter_id` BIGINT UNSIGNED NOT NULL COMMENT '邀请者ID（关联sys_user.user_id，通常是房主）',
  `invitee_id` BIGINT UNSIGNED NOT NULL COMMENT '被邀请者ID（关联sys_user.user_id）',
  `status` ENUM('PENDING', 'ACCEPTED', 'FINISHED') NOT NULL DEFAULT 'PENDING' COMMENT '邀请状态：待接受/已接受/已结束',
  `create_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '邀请创建时间',
  PRIMARY KEY (`invitation_id`),
  INDEX `idx_room_id` (`room_id`),
  INDEX `idx_invitee_id` (`invitee_id`),
  -- 外键约束
  CONSTRAINT `fk_chat_game_invitations_room`
    FOREIGN KEY (`room_id`) REFERENCES `chat_games` (`room_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_chat_game_invitations_inviter`
    FOREIGN KEY (`inviter_id`) REFERENCES `sys_user` (`user_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_chat_game_invitations_invitee`
    FOREIGN KEY (`invitee_id`) REFERENCES `sys_user` (`user_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='游戏房间邀请记录表';

CREATE TABLE IF NOT EXISTS `chat_game_members`(
  `member_id` BIGINT UNSIGNED NOT NULL COMMENT '成员ID（关联sys_user）',
  `room_id` VARCHAR(36) NOT NULL COMMENT '房间ID（关联chat_games）',
  `status` ENUM('ONLINE', 'SUSPENDED', 'READY', 'OFFLINE', 'IN_GAME') NOT NULL DEFAULT 'ONLINE' COMMENT '成员状态：在线/挂起/已准备/离线/游戏中',
  `join_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '加入时间',
  PRIMARY KEY (`member_id`, `room_id`), -- 复合主键（用户+房间唯一）
  INDEX `idx_room_id` (`room_id`),
  CONSTRAINT `fk_chat_room_members_member` FOREIGN KEY (`member_id`) REFERENCES `sys_user` (`user_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_chat_room_members_room` FOREIGN KEY (`room_id`) REFERENCES `chat_games` (`room_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='聊天室成员表';

CREATE TABLE IF NOT EXISTS `chat_game_messages` (
  `message_id` VARCHAR(36) NOT NULL COMMENT '消息唯一ID（UUID）',
  `room_id` VARCHAR(36) NOT NULL COMMENT '所属游戏房间ID（关联chat_games.room_id）',
  `sender_id` BIGINT UNSIGNED NOT NULL COMMENT '发送者ID（关联sys_user.user_id）',
  `content` LONGTEXT NOT NULL COMMENT '消息内容（文本/图片URL/文件路径等）',
  `message_type` ENUM('TEXT', 'IMAGE', 'FILE', 'VOICE', 'VIDEO') NOT NULL DEFAULT 'TEXT' COMMENT '消息类型',
  `status` ENUM('SENT', 'FAILED', 'RETRACTED') NOT NULL DEFAULT 'SENT' COMMENT '消息状态：已发送/发送失败/已撤回',
  `create_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '消息发送时间',
  PRIMARY KEY (`message_id`),
  -- 索引优化查询
  INDEX `idx_room_id` (`room_id`),       -- 按游戏房间查消息
  INDEX `idx_sender_id` (`sender_id`),   -- 按发送者查消息
  INDEX `idx_create_time` (`create_time`),-- 按时间排序
  -- 外键约束
  CONSTRAINT `fk_chat_game_messages_room`
    FOREIGN KEY (`room_id`) REFERENCES `chat_games` (`room_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_chat_game_messages_sender`
    FOREIGN KEY (`sender_id`) REFERENCES `sys_user` (`user_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='游戏大厅消息表';


CREATE TABLE IF NOT EXISTS `chat_game_message_reads` (
  `message_id` VARCHAR(36) NOT NULL COMMENT '关联的游戏大厅消息ID（chat_game_messages.message_id）',
  `member_id` BIGINT UNSIGNED NOT NULL COMMENT '已读成员ID（关联sys_user.user_id）',
  `read_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '已读时间',
  PRIMARY KEY (`message_id`, `member_id`), -- 复合主键（消息+成员唯一）
  -- 索引优化已读查询
  INDEX `idx_member_id` (`member_id`),     -- 按成员查已读消息
  -- 外键约束
  CONSTRAINT `fk_chat_game_message_reads_message`
    FOREIGN KEY (`message_id`) REFERENCES `chat_game_messages` (`message_id`) ON DELETE CASCADE,
  CONSTRAINT `fk_chat_game_message_reads_member`
    FOREIGN KEY (`member_id`) REFERENCES `sys_user` (`user_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='游戏大厅消息已读记录表';
