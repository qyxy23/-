-- 海龟汤系统数据库表结构（混合方案版本）

-- 核心海龟汤表（扩展现有设计）
CREATE TABLE IF NOT EXISTS `hai_gui_soup` (
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤唯一ID（UUID）',
  `soup_title` VARCHAR(255) DEFAULT '' COMMENT '汤面标题（如"双鱼玉佩的游戏"）',
  `soup_surface` TEXT NOT NULL COMMENT '汤面文本（玩家看到的悬念故事）',
  `soup_bottom` TEXT NOT NULL COMMENT '汤底文本（完整故事逻辑：背景、多视角、主持人手册）',
  `host_manual` TEXT NOT NULL COMMENT '主持人手册（推理规则、进度提示）',
  `key_clues` JSON NOT NULL COMMENT '关键线索ID列表（JSON数组，如["clue_123", "clue_456"]，对应hai_gui_soup_clue.clue_id）',
  `soup_surface_vec` VARCHAR(255) DEFAULT '' COMMENT '汤面向量的Redis Stack键名（如"hai_gui:vec:surface:xxx"）',
  `soup_bottom_vec` VARCHAR(255) DEFAULT '' COMMENT '汤底向量的Redis Stack键名',

  -- 混合方案新增字段
  `task_generation_strategy` VARCHAR(20) DEFAULT 'HYBRID' COMMENT '任务生成策略：HYBRID-混合方案，DYNAMIC-纯AI，STATIC-静态预设',
  `clue_decomposition_config` JSON COMMENT '线索拆解配置（AI生成的线索片段列表）',
  `vector_match_threshold` DECIMAL(3,2) DEFAULT 0.7 COMMENT '向量匹配阈值',

  `creator_id` BIGINT UNSIGNED COMMENT '实际创作者ID（外键：sys_user.user_id）',
  `uploader_id` BIGINT UNSIGNED COMMENT '上传者ID（外键：sys_user.user_id）',
  `upload_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '上传时间',
  `play_count` INT NOT NULL DEFAULT 0 COMMENT '游玩次数',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记（0=未删，1=已删）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  PRIMARY KEY (`soup_id`),
  INDEX `idx_is_deleted` (`is_deleted`),
  INDEX `idx_created_at` (`created_at`),
  INDEX `idx_key_clues` ((CAST(key_clues AS CHAR(255) ARRAY))), -- MySQL 8.0+ JSON数组索引优化
  INDEX `idx_generation_strategy` (`task_generation_strategy`),

  CONSTRAINT `fk_hai_gui_soup_creator`
    FOREIGN KEY (`creator_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE SET NULL,
  CONSTRAINT `fk_hai_gui_soup_uploader`
    FOREIGN KEY (`uploader_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='海龟汤核心定义表（混合方案版）';


-- 拆解线索表（用于向量匹配）
CREATE TABLE IF NOT EXISTS `hai_gui_soup_clue_fragment` (
  `fragment_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '线索片段ID',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤ID',
  `fragment_content` VARCHAR(500) NOT NULL COMMENT '线索片段内容（拆解后的汤底片段）',
  `fragment_type` VARCHAR(50) NOT NULL COMMENT '片段类型：TIME-时间，PLACE-地点，CHARACTER-人物，PLOT-情节，OBJECT-物品，TRUTH-真相',
  `inference_level` INT DEFAULT 1 COMMENT '推理深度：1-表层信息，2-中层推理，3-深层真相，4-核心逻辑',
  `vector_hash` VARCHAR(64) COMMENT '向量数据的哈希值（用于缓存）',
  `vector_data` JSON COMMENT '向量数据（BGE向量）',
  `associated_task_ids` JSON COMMENT '关联的任务ID列表',
  `trigger_keywords` JSON COMMENT '触发关键词列表',
  `similarity_threshold` DECIMAL(3,2) DEFAULT 0.7 COMMENT '该片段的匹配阈值',
  `is_core_clue` TINYINT(1) DEFAULT 0 COMMENT '是否为核心线索（直接影响最终真相）',
  `fragment_order` INT DEFAULT 0 COMMENT '片段在逻辑链中的顺序',
  `generation_source` VARCHAR(20) DEFAULT 'AI' COMMENT '生成来源：AI-AI拆解，MANUAL-手动拆解',
  `ai_analysis_confidence` DECIMAL(3,2) COMMENT 'AI分析的置信度（0-1）',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  FOREIGN KEY (`soup_id`) REFERENCES `hai_gui_soup`(`soup_id`) ON DELETE CASCADE,
  INDEX `idx_soup_id` (`soup_id`),
  INDEX `idx_fragment_type` (`fragment_type`),
  INDEX `idx_inference_level` (`inference_level`),
  INDEX `idx_vector_hash` (`vector_hash`),
  INDEX `idx_core_clue` (`is_core_clue`),
  INDEX `idx_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='拆解线索表（向量化用）';


-- 推理任务表（重新设计，基于理解层次）
CREATE TABLE IF NOT EXISTS `hai_gui_soup_inference_task` (
  `task_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '推理任务ID',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤ID',
  `task_name` VARCHAR(100) NOT NULL COMMENT '任务名称',
  `task_description` TEXT NOT NULL COMMENT '任务描述（要达到的理解层次）',
  `understanding_level` INT NOT NULL COMMENT '理解层次：1-发现表层事实，2-理解内在联系，3-推理部分真相，4-掌握核心逻辑',
  `target_keywords` JSON COMMENT '目标关键词（玩家能说出这些词才算理解）',
  `reasoning_goal` TEXT NOT NULL COMMENT '推理目标（AI判断任务的完成标准）',
  `progress_weight` DECIMAL(5,2) NOT NULL COMMENT '进度权重',
  `required_fragments` JSON COMMENT '需要关联的线索片段ID',
  `evaluation_prompt` TEXT COMMENT 'AI评估任务完成度的专用提示词',
  `is_mandatory` TINYINT(1) DEFAULT 1 COMMENT '是否为必做任务（核心逻辑必须理解）',
  `task_order` INT DEFAULT 0 COMMENT '任务逻辑顺序',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  FOREIGN KEY (`soup_id`) REFERENCES `hai_gui_soup`(`soup_id`) ON DELETE CASCADE,
  INDEX `idx_soup_id` (`soup_id`),
  INDEX `idx_understanding_level` (`understanding_level`),
  INDEX `idx_task_order` (`task_order`),
  INDEX `idx_mandatory` (`is_mandatory`),
  INDEX `idx_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='推理任务表（基于理解层次设计）';


-- 游戏会话表（保持原有设计）
CREATE TABLE IF NOT EXISTS `haigui_game_session` (
  `session_id` VARCHAR(36) NOT NULL COMMENT '游戏会话唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '发起游戏的玩家ID（外键：sys_user.user_id）',
  `chat_session_id` VARCHAR(36) NOT NULL COMMENT '关联的聊天会话ID（外键：ai_chat_sessions.session_id）',
  `start_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '游戏开始时间',
  `end_time` DATETIME(6) DEFAULT NULL COMMENT '游戏结束时间（NULL=进行中）',
  `current_progress` DECIMAL(5,2) NOT NULL DEFAULT 0.00 COMMENT '当前进度百分比（如35.50）',
  `status` ENUM('ONGOING','COMPLETED','CANCELED') NOT NULL DEFAULT 'ONGOING' COMMENT '游戏状态',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `score` DECIMAL(5,2) DEFAULT 0.00 COMMENT '游戏最终得分（根据进度/线索完成度计算）',

  PRIMARY KEY (`session_id`),
  INDEX `idx_soup_id` (`soup_id`),
  INDEX `idx_user_id` (`user_id`),
  INDEX `idx_chat_session_id` (`chat_session_id`),
  INDEX `idx_status` (`status`),

  CONSTRAINT `fk_game_session_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_game_session_user`
    FOREIGN KEY (`user_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_game_session_chat`
    FOREIGN KEY (`chat_session_id`)
    REFERENCES `ai_chat_sessions` (`session_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='海龟汤游戏会话表';


-- 用户推理进度表（重新设计）
CREATE TABLE IF NOT EXISTS `hai_gui_user_inference_progress` (
  `progress_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '推理进度ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤ID',
  `session_id` VARCHAR(36) NOT NULL COMMENT '游戏会话ID',
  `task_id` BIGINT NOT NULL COMMENT '推理任务ID',
  `understanding_progress` DECIMAL(5,2) DEFAULT 0.0 COMMENT '理解进度百分比（0-100）',
  `last_question` TEXT COMMENT '最后的问题',
  `ai_evaluation` JSON COMMENT 'AI评估结果：{"score": 85, "reason": "...", "keywords_matched": ["星际旅行"]}',
  `discovered_fragments` JSON COMMENT '已发现的线索片段ID',
  `target_keywords_matched` JSON COMMENT '已匹配的目标关键词',
  `is_completed` TINYINT(1) DEFAULT 0 COMMENT '是否完成该推理任务',
  `completion_time` DATETIME(6) DEFAULT NULL COMMENT '完成时间',
  `questions_count` INT DEFAULT 0 COMMENT '针对该任务的问题数量',
  `evaluation_history` JSON COMMENT '评估历史（记录每次AI评估结果）',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  FOREIGN KEY (`soup_id`) REFERENCES `hai_gui_soup`(`soup_id`) ON DELETE CASCADE,
  FOREIGN KEY (`session_id`) REFERENCES `haigui_game_session`(`session_id`) ON DELETE CASCADE,
  FOREIGN KEY (`task_id`) REFERENCES `hai_gui_soup_inference_task`(`task_id`) ON DELETE CASCADE,
  UNIQUE KEY `uk_user_soup_task` (`user_id`, `soup_id`, `task_id`),
  INDEX `idx_user_soup` (`user_id`, `soup_id`),
  INDEX `idx_understanding_progress` (`understanding_progress`),
  INDEX `idx_completion_time` (`completion_time`),
  INDEX `idx_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户推理进度表';


-- 问题处理记录表（混合方案核心）
CREATE TABLE IF NOT EXISTS `hai_gui_question_processing_log` (
  `log_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '处理记录ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤ID',
  `session_id` VARCHAR(36) NOT NULL COMMENT '游戏会话ID',
  `question_text` TEXT NOT NULL COMMENT '用户问题原文',

  -- 向量匹配结果
  `question_vector` VARCHAR(1000) COMMENT '问题的BGE向量',
  `matched_fragments` JSON COMMENT '向量匹配到的线索片段列表：[{"id": 1, "similarity": 0.85, "content": "..."}]',
  `match_threshold` DECIMAL(3,2) COMMENT '实际使用的匹配阈值',
  `match_time_ms` BIGINT COMMENT '向量匹配耗时（毫秒）',

  -- AI判断结果
  `ai_evaluation_prompt` TEXT COMMENT '发送给AI的完整提示词',
  `ai_evaluation_response` JSON COMMENT 'AI的评估结果',
  `ai_evaluation_time_ms` BIGINT COMMENT 'AI评估耗时（毫秒）',

  -- 结果汇总
  `affected_tasks` JSON COMMENT '影响的推理任务：[{"taskId": 1, "progressChange": +25, "isCompleted": false}]',
  `final_response` TEXT COMMENT '返回给用户的最终回答',
  `total_processing_time_ms` BIGINT COMMENT '总处理时间（毫秒）',

  -- 缓存和优化
  `response_cache_key` VARCHAR(128) COMMENT '响应缓存键（用于类似问题复用）',
  `question_fingerprint` VARCHAR(64) COMMENT '问题指纹（用于去重）',

  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',

  FOREIGN KEY (`soup_id`) REFERENCES `hai_gui_soup`(`soup_id`) ON DELETE CASCADE,
  FOREIGN KEY (`session_id`) REFERENCES `haigui_game_session`(`session_id`) ON DELETE CASCADE,
  INDEX `idx_user_soup` (`user_id`, `soup_id`),
  INDEX `idx_session_id` (`session_id`),
  INDEX `idx_question_fingerprint` (`question_fingerprint`),
  INDEX `idx_response_cache_key` (`response_cache_key`),
  INDEX `idx_created_at` (`created_at`),
  INDEX `idx_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='问题处理记录表（混合方案核心）';


-- 线索片段缓存表（Redis Stack的备份）
CREATE TABLE IF NOT EXISTS `hai_gui_clue_fragment_cache` (
  `cache_id` BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '缓存ID',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤ID',
  `fragment_id` BIGINT NOT NULL COMMENT '线索片段ID',
  `cache_key` VARCHAR(128) NOT NULL COMMENT 'Redis中的缓存键',
  `vector_data` JSON NOT NULL COMMENT '向量数据',
  `cache_version` INT DEFAULT 1 COMMENT '缓存版本',
  `hit_count` INT DEFAULT 0 COMMENT '命中次数',
  `last_access_time` DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '最后访问时间',
  `expires_at` DATETIME(6) COMMENT '过期时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',

  FOREIGN KEY (`soup_id`) REFERENCES `hai_gui_soup`(`soup_id`) ON DELETE CASCADE,
  FOREIGN KEY (`fragment_id`) REFERENCES `hai_gui_soup_clue_fragment`(`fragment_id`) ON DELETE CASCADE,
  UNIQUE KEY `uk_cache_key` (`cache_key`),
  INDEX `idx_soup_fragment` (`soup_id`, `fragment_id`),
  INDEX `idx_expires_at` (`expires_at`),
  INDEX `idx_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='线索片段缓存表（Redis Stack备份）';


-- 向量元数据表（保持原有设计）
CREATE TABLE IF NOT EXISTS `haigui_vector_metadata` (
  `vector_id` VARCHAR(36) NOT NULL COMMENT '向量唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `vector_type` ENUM('SURFACE','BOTTOM','MANUAL','CLUE','FRAGMENT') NOT NULL COMMENT '向量类型：汤面/汤底/主持人手册/线索/线索片段',
  `redis_key` VARCHAR(255) NOT NULL COMMENT 'Redis Stack中的向量键名（如"hai_gui:vec:surface:xxx"）',
  `vector_dim` INT NOT NULL COMMENT '向量维度（如768=对应BGE-small-zh-v1.5）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '向量创建时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',

  PRIMARY KEY (`vector_id`),
  INDEX `idx_soup_id` (`soup_id`),
  INDEX `idx_vector_type` (`vector_type`),

  CONSTRAINT `fk_vector_metadata_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='向量元数据表（关联Redis Stack）';
