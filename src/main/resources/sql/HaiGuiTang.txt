CREATE TABLE IF NOT EXISTS `hai_gui_soup` (
  `soup_id` VARCHAR(36) NOT NULL COMMENT '海龟汤唯一ID（UUID）',
  `soup_title` VARCHAR(255) DEFAULT '' COMMENT '汤面标题（如“双鱼玉佩的游戏”）',
  `soup_surface` TEXT NOT NULL COMMENT '汤面文本（玩家看到的悬念故事）',
  `soup_bottom` TEXT NOT NULL COMMENT '汤底文本（完整故事逻辑：背景、多视角、主持人手册）',
  `host_manual` TEXT NOT NULL COMMENT '主持人手册（推理规则、进度提示）',
  -- ✅ 调整为存储线索ID数组
  `key_clues` JSON NOT NULL COMMENT '关键线索ID列表（JSON数组，如["clue_123", "clue_456"]，对应haigui_soup_clue.clue_id）',
  `progress_settings` JSON NOT NULL COMMENT '进度规则（JSON对象，如{"MOM_IDENTITY":15,"WIFE_CLONE":20}，定义各进度类型的增量）',
  `soup_surface_vec` VARCHAR(255) DEFAULT '' COMMENT '汤面向量的Redis Stack键名（如"hai_gui:vec:surface:xxx"）',
  `soup_bottom_vec` VARCHAR(255) DEFAULT '' COMMENT '汤底向量的Redis Stack键名',
  `creator_id` BIGINT UNSIGNED COMMENT '实际创作者ID（外键：sys_user.user_id）',
  `uploader_id` BIGINT UNSIGNED COMMENT '上传者ID（外键：sys_user.user_id）',
  `upload_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '上传时间',
  `play_count` INT NOT NULL DEFAULT 0 COMMENT '游玩次数',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记（0=未删，1=已删）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  PRIMARY KEY (`soup_id`),
  INDEX `idx_is_deleted` (`is_deleted`),
  INDEX `idx_created_at` (`created_at`),
  -- 新增索引：加速按线索ID查询（若需）
  INDEX `idx_key_clues` ((CAST(key_clues AS CHAR(255) ARRAY))), -- MySQL 8.0+支持JSON数组的索引优化

  CONSTRAINT `fk_hai_gui_soup_creator`
    FOREIGN KEY (`creator_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE SET NULL,
  CONSTRAINT `fk_hai_gui_soup_uploader`
    FOREIGN KEY (`uploader_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='海龟汤核心定义表';





CREATE TABLE IF NOT EXISTS `haigui_game_session` (
  `session_id` VARCHAR(36) NOT NULL COMMENT '游戏会话唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '发起游戏的玩家ID（外键：sys_user.user_id）',
  `chat_session_id` VARCHAR(36) NOT NULL COMMENT '关联的聊天会话ID（外键：ai_chat_sessions.session_id）',
  `start_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '游戏开始时间',
  `end_time` DATETIME(6) DEFAULT NULL COMMENT '游戏结束时间（NULL=进行中）',
  `current_progress` DECIMAL(5,2) NOT NULL DEFAULT 0.00 COMMENT '当前进度百分比（如35.50）',
  `status` ENUM('ONGOING','COMPLETED','CANCELED') NOT NULL DEFAULT 'ONGOING' COMMENT '游戏状态',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',

  PRIMARY KEY (`session_id`),
  INDEX `idx_soup_id` (`soup_id`), -- 按海龟汤筛选会话
  INDEX `idx_user_id` (`user_id`), -- 按玩家筛选会话
  INDEX `idx_chat_session_id` (`chat_session_id`), -- 关联聊天会话
  INDEX `idx_status` (`status`), -- 快速筛选进行中的游戏

  CONSTRAINT `fk_game_session_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_game_session_user`
    FOREIGN KEY (`user_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_game_session_chat`
    FOREIGN KEY (`chat_session_id`)
    REFERENCES `ai_chat_sessions` (`session_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='海龟汤游戏会话表';



CREATE TABLE IF NOT EXISTS `haigui_soup_clue` (
  `clue_id` VARCHAR(36) NOT NULL COMMENT '线索唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `clue_content` TEXT NOT NULL COMMENT '线索内容（如“1978年用望远镜看深渊，下面是雪地（实际是复制体白骨）”）',
  `clue_type` ENUM('TIME','PLACE','CHARACTER','PLOT') NOT NULL COMMENT '线索类型：时间/地点/人物/剧情',
  `is_key` TINYINT(1) NOT NULL DEFAULT 1 COMMENT '是否是关键线索（1=是，影响进度；0=否）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '线索创建时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',

  PRIMARY KEY (`clue_id`),
  INDEX `idx_soup_id` (`soup_id`), -- 按海龟汤筛选线索
  INDEX `idx_clue_type` (`clue_type`), -- 按线索类型筛选
  INDEX `idx_is_key` (`is_key`), -- 快速筛选关键线索

  CONSTRAINT `fk_soup_clue_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='海龟汤线索表';



CREATE TABLE IF NOT EXISTS `haigui_player_progress` (
  `progress_id` VARCHAR(36) NOT NULL COMMENT '进度记录唯一ID（UUID）',
  `session_id` VARCHAR(36) NOT NULL COMMENT '关联的游戏会话ID（外键：haigui_game_session.session_id）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `progress_key` VARCHAR(50) NOT NULL COMMENT '进度类型（如"MOM_IDENTITY"=妈妈是复制体，"WIFE_CLONE"=妻子是复制体）',
  `progress_value` DECIMAL(5,2) NOT NULL COMMENT '进度增量（如15.00=+15%）',
  `clue_id` VARCHAR(36) DEFAULT NULL COMMENT '关联的线索ID（外键：haigui_soup_clue.clue_id，可选）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '进度更新时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',

  PRIMARY KEY (`progress_id`),
  INDEX `idx_session_id` (`session_id`), -- 按游戏会话筛选进度
  INDEX `idx_soup_id` (`soup_id`), -- 按海龟汤筛选进度
  INDEX `idx_progress_key` (`progress_key`), -- 按进度类型筛选

  CONSTRAINT `fk_player_progress_session`
    FOREIGN KEY (`session_id`)
    REFERENCES `haigui_game_session` (`session_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_player_progress_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_player_progress_clue`
    FOREIGN KEY (`clue_id`)
    REFERENCES `haigui_soup_clue` (`clue_id`)
    ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='玩家进度跟踪表';



CREATE TABLE IF NOT EXISTS `haigui_vector_metadata` (
  `vector_id` VARCHAR(36) NOT NULL COMMENT '向量唯一ID（UUID）',
  `soup_id` VARCHAR(36) NOT NULL COMMENT '关联的海龟汤ID（外键：hai_gui_soup.soup_id）',
  `vector_type` ENUM('SURFACE','BOTTOM','MANUAL') NOT NULL COMMENT '向量类型：汤面/汤底/主持人手册',
  `redis_key` VARCHAR(255) NOT NULL COMMENT 'Redis Stack中的向量键名（如"hai_gui:vec:surface:xxx"）',
  `vector_dim` INT NOT NULL COMMENT '向量维度（如768=对应BGE-small-zh-v1.5）',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '向量创建时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记',

  PRIMARY KEY (`vector_id`),
  INDEX `idx_soup_id` (`soup_id`), -- 按海龟汤筛选向量
  INDEX `idx_vector_type` (`vector_type`), -- 按向量类型筛选

  CONSTRAINT `fk_vector_metadata_soup`
    FOREIGN KEY (`soup_id`)
    REFERENCES `hai_gui_soup` (`soup_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='向量元数据表（关联Redis Stack）';


CREATE TABLE IF NOT EXISTS `haigui_player_ai_dialog_stats` (
  `stat_id` VARCHAR(36) NOT NULL COMMENT '统计记录唯一ID（UUID）',
  `session_id` VARCHAR(36) NOT NULL COMMENT '关联的游戏会话ID（外键：haigui_game_session.session_id）',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '玩家ID（外键：sys_user.user_id）',
  `dialog_count` INT NOT NULL DEFAULT 0 COMMENT '与AI的总对话次数',
  `yes_count` INT NOT NULL DEFAULT 0 COMMENT 'AI回答“是”的次数',
  `created_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间',
  `updated_at` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '更新时间',

  PRIMARY KEY (`stat_id`),
  INDEX `idx_session_id` (`session_id`), -- 按游戏会话筛选
  INDEX `idx_user_id` (`user_id`), -- 按玩家筛选

  CONSTRAINT `fk_player_ai_dialog_stats_session`
    FOREIGN KEY (`session_id`)
    REFERENCES `haigui_game_session` (`session_id`)
    ON DELETE CASCADE,
  CONSTRAINT `fk_player_ai_dialog_stats_user`
    FOREIGN KEY (`user_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='玩家与AI对话统计表';
