CREATE TABLE IF NOT EXISTS `ai_chat_sessions` (
  `session_id` VARCHAR(36) NOT NULL COMMENT '会话唯一ID（UUID）',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '发起AI聊天的用户ID（关联sys_user）',
  `title` VARCHAR(255) NOT NULL DEFAULT '新对话' COMMENT '会话标题（如“群聊AI助手”）',
  `create_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '会话创建时间',
  `update_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '会话最后更新时间',
  `is_deleted` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '逻辑删除标记（0=未删，1=已删）',

  -- 🔑 新增：关联不同聊天场景的核心字段
  `context_type` ENUM('GROUP_CHAT', 'PRIVATE_CHAT', 'GAME_ROOM', 'AI_SOLO') NOT NULL COMMENT '上下文类型：
    GROUP_CHAT-群聊AI
    PRIVATE_CHAT-私聊AI
    GAME_ROOM-游戏大厅AI
    AI_SOLO-纯AI单独对话',
  `context_id` VARCHAR(36) NOT NULL COMMENT '上下文ID：对应场景的唯一标识（如群聊group_id/游戏room_id/私聊用户ID）',

  PRIMARY KEY (`session_id`),
  INDEX `idx_user_id` (`user_id`),          -- 按用户查会话
  INDEX `idx_create_time` (`create_time`),  -- 按时间查会话
  INDEX `idx_context` (`context_type`, `context_id`), -- 按场景查会话（关键索引）
  CONSTRAINT `fk_ai_chat_sessions_user`
    FOREIGN KEY (`user_id`)
    REFERENCES `sys_user` (`user_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI聊天会话表（支持多场景）';

-- 保持原表结构，仅需规范sender_id的取值：
-- - sender_type=ASSISTANT时，sender_id填AI的唯一标识（如AI模型ID/名称，例："AI_Assistant_v1"）；
-- - sender_type=USER时，sender_id填用户ID；
-- - sender_type=SYSTEM时，sender_id填系统标识（如"SYSTEM_NOTICE"）。
CREATE TABLE IF NOT EXISTS `ai_chat_messages` (
  `msg_id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '消息ID',
  `session_id` VARCHAR(36) NOT NULL COMMENT '关联的AI会话ID',
  `sender_type` ENUM('USER', 'ASSISTANT', 'SYSTEM') NOT NULL COMMENT '发送者类型',
  -- 调整sender_id类型为VARCHAR，兼容不同场景的取值（用户ID/ AI标识/系统标识）
  `sender_id` VARCHAR(255) NOT NULL COMMENT '发送者ID（规则：USER=用户ID；ASSISTANT=AI模型ID/名称；SYSTEM=系统标识）',
  `content` TEXT NOT NULL COMMENT '消息内容',
  `send_time` DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '发送时间',
  `is_read` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否已读（仅用户未读AI消息有效）',
  `is_yes_answer` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否是AI回答“是”（1=是，0=否）',

  -- 主键与索引
  PRIMARY KEY (`msg_id`),
  INDEX `idx_session_time` (`session_id`, `send_time`), -- 按会话+时间查消息
  INDEX `idx_sender_time` (`sender_id`, `send_time`),   -- 按发送者+时间查消息

  -- 仅保留有效的外键约束（session_id关联会话表）
  CONSTRAINT `fk_ai_chat_messages_session`
    FOREIGN KEY (`session_id`)
    REFERENCES `ai_chat_sessions` (`session_id`)
    ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI聊天消息表';

-- 1. 添加 BEFORE INSERT 触发器
DELIMITER $$
CREATE TRIGGER trg_check_user_sender
BEFORE INSERT ON ai_chat_messages
FOR EACH ROW
BEGIN
    IF NEW.sender_type = 'USER' AND NEW.sender_id IS NULL THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'USER 类型消息必须指定 sender_id';
    END IF;
END
$$
DELIMITER ;


-- 2. 添加 BEFORE UPDATE 触发器
DELIMITER $$
CREATE TRIGGER trg_update_user_sender
BEFORE UPDATE ON ai_chat_messages
FOR EACH ROW
BEGIN
    IF NEW.sender_type = 'USER' AND NEW.sender_id IS NULL THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'USER 类型消息必须指定 sender_id';
    END IF;
END
$$
DELIMITER ;
